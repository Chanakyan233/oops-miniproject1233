CREATE TABLE IF NOT EXISTS MaintenanceRecords (
    record_id INT AUTO_INCREMENT PRIMARY KEY,
    car_model VARCHAR(100),
    owner_name VARCHAR(100),
    maintenance_task VARCHAR(100),
    service_date DATE,
    status VARCHAR(50)
);
